        function getData() {

    $.ajax({
		url: 'index.aspx/GetDataAll',
        type: 'POST',
        contentType: 'application/json; charset=utf-8',
        dataType: 'json',
    }).done(function (res) {
        let tb = $("#tbInfo");

        tb.empty()

        $.each(res.d, function (index, item) {
            let html = `<tr>` +
                `<td>` + item.MACP + `</td>` +
                `<td>` + item.GMUA3 + `</td>` +
                `<td>` + item.SLMUA3 + `</td>` +
                `<td>` + item.GMUA2 + `</td>` +
                `<td>` + item.SLMUA2 + `</td>` +
                `<td>` + item.GMUA1 + `</td>` +
                `<td>` + item.SLMUA1 + `</td>` +
                `<td>` + item.GIAKHOP + `</td>` +
                `<td>` + item.SOLUONGKHOP + `</td>` +
                `<td>` + item.GBAN1 + `</td>` +
                `<td>` + item.SLBAN1 + `</td>` +
                `<td>` + item.GBAN2 + `</td>` +
                `<td>` + item.SLBAN2 + `</td>` +
                `<td>` + item.GBAN3 + `</td>` +
                `<td>` + item.SLBAN3 + `</td>` +
                `<td>` + item.TONGSLKHOP + `</td>` +
                `</tr>`
            tb.append(html)

        })
    }).fail(function () {

    })

}


function getConnectHub() {
    let tb = $.connection.myHub;

    tb.client.displayStatus = function () {
        getData()
	}

    $.connection.hub.start()

    getData()

}
